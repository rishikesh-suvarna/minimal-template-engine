MinimalTemplate - Basic Example

Simple Variable Example

const template = ` <div class="user-card"> <h3>Hello, {{ name }}!</h3> <p>Your role: {{ role }}</p> </div> `; const data = { name: "John Smith", role: "Developer" }; engine.renderToDOM('#simple-example', template, data);

Conditional Rendering Example

const template = ` <div class="product-info"> <h3>{{ product.name }}</h3> <p>{{ product.description }}</p> <p>Price: ${{ product.price }}</p> {{ if product.inStock }} <button class="buy-btn">Add to Cart</button> {{ else }} <p class="out-of-stock">Currently Out of Stock</p> {{ /if }} </div> `; const data = { product: { name: "Smart Watch", description: "Latest smartwatch with health tracking features", price: 249.99, inStock: true } }; engine.renderToDOM('#conditional-example', template, data);
